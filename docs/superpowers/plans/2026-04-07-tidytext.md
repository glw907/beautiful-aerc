// internal/tidy/config.go
package tidy

import (
	"errors"
	"fmt"
	"os"
	"strings"

	"github.com/BurntSushi/toml"
)

type APIConfig struct {
	Model  string `toml:"model"`
	APIKey string `toml:"api_key"`
}

type RulesConfig struct {
	Spelling           bool   `toml:"spelling"`
	Grammar            bool   `toml:"grammar"`
	Punctuation        bool   `toml:"punctuation"`
	Whitespace         bool   `toml:"whitespace"`
	Capitalization     bool   `toml:"capitalization"`
	RepeatedWords      bool   `toml:"repeated_words"`
	MissingPunctuation bool   `toml:"missing_punctuation"`
	OxfordComma        string `toml:"oxford_comma"`
}

type StyleConfig struct {
	EmDashSpaces       bool     `toml:"em_dash_spaces"`
	Ellipsis           string   `toml:"ellipsis"`
	CustomInstructions []string `toml:"custom_instructions"`
}

type Config struct {
	API   APIConfig   `toml:"api"`
	Rules RulesConfig `toml:"rules"`
	Style StyleConfig `toml:"style"`
}

func DefaultConfig() Config {
	return Config{
		API: APIConfig{
			Model: "claude-haiku-4-5-20251001",
		},
		Rules: RulesConfig{
			Spelling:           true,
			Grammar:            true,
			Punctuation:        true,
			Whitespace:         true,
			Capitalization:     true,
			RepeatedWords:      true,
			MissingPunctuation: true,
			OxfordComma:        "ignore",
		},
		Style: StyleConfig{
			EmDashSpaces: false,
			Ellipsis:     "character",
		},
	}
}

func LoadConfig(path string) (Config, error) {
	cfg := DefaultConfig()

	data, err := os.ReadFile(path)
	if errors.Is(err, os.ErrNotExist) {
		return cfg, nil
	}
	if err != nil {
		return cfg, fmt.Errorf("reading config %s: %w", path, err)
	}

	if _, err := toml.Decode(string(data), &cfg); err != nil {
		return cfg, fmt.Errorf("parsing config %s: %w", path, err)
	}

	if err := validateConfig(cfg); err != nil {
		return cfg, fmt.Errorf("config %s: %w", path, err)
	}

	return cfg, nil
}

func validateConfig(cfg Config) error {
	switch cfg.Rules.OxfordComma {
	case "insert", "remove", "ignore":
	default:
		return fmt.Errorf("invalid oxford_comma value %q (want insert, remove, or ignore)", cfg.Rules.OxfordComma)
	}

	switch cfg.Style.Ellipsis {
	case "character", "dots":
	default:
		return fmt.Errorf("invalid ellipsis value %q (want character or dots)", cfg.Style.Ellipsis)
	}

	return nil
}

func ApplyRuleOverrides(cfg *Config, overrides []string) error {
	for _, o := range overrides {
		key, value, ok := strings.Cut(o, "=")
		if !ok {
			return fmt.Errorf("invalid rule override %q (expected key=value)", o)
		}
		switch key {
		case "spelling":
			cfg.Rules.Spelling = value == "true"
		case "grammar":
			cfg.Rules.Grammar = value == "true"
		case "punctuation":
			cfg.Rules.Punctuation = value == "true"
		case "whitespace":
			cfg.Rules.Whitespace = value == "true"
		case "capitalization":
			cfg.Rules.Capitalization = value == "true"
		case "repeated_words":
			cfg.Rules.RepeatedWords = value == "true"
		case "missing_punctuation":
			cfg.Rules.MissingPunctuation = value == "true"
		case "oxford_comma":
			cfg.Rules.OxfordComma = value
		default:
			return fmt.Errorf("unknown rule %q", key)
		}
	}
	return validateConfig(*cfg)
}

func ApplyStyleOverrides(cfg *Config, overrides []string) error {
	for _, o := range overrides {
		key, value, ok := strings.Cut(o, "=")
		if !ok {
			return fmt.Errorf("invalid style override %q (expected key=value)", o)
		}
		switch key {
		case "em_dash_spaces":
			cfg.Style.EmDashSpaces = value == "true"
		case "ellipsis":
			cfg.Style.Ellipsis = value
		default:
			return fmt.Errorf("unknown style %q", key)
		}
	}
	return validateConfig(*cfg)
}

func ResolveAPIKey(cfg Config) string {
	if cfg.API.APIKey != "" {
		return cfg.API.APIKey
	}
	return os.Getenv("ANTHROPIC_API_KEY")
}

func ConfigString(cfg Config) string {
	var b strings.Builder
	b.WriteString("[api]\n")
	b.WriteString(fmt.Sprintf("model = %q\n", cfg.API.Model))
	b.WriteString("\n[rules]\n")
	b.WriteString(fmt.Sprintf("spelling = %t\n", cfg.Rules.Spelling))
	b.WriteString(fmt.Sprintf("grammar = %t\n", cfg.Rules.Grammar))
	b.WriteString(fmt.Sprintf("punctuation = %t\n", cfg.Rules.Punctuation))
	b.WriteString(fmt.Sprintf("whitespace = %t\n", cfg.Rules.Whitespace))
	b.WriteString(fmt.Sprintf("capitalization = %t\n", cfg.Rules.Capitalization))
	b.WriteString(fmt.Sprintf("repeated_words = %t\n", cfg.Rules.RepeatedWords))
	b.WriteString(fmt.Sprintf("missing_punctuation = %t\n", cfg.Rules.MissingPunctuation))
	b.WriteString(fmt.Sprintf("oxford_comma = %q\n", cfg.Rules.OxfordComma))
	b.WriteString("\n[style]\n")
	b.WriteString(fmt.Sprintf("em_dash_spaces = %t\n", cfg.Style.EmDashSpaces))
	b.WriteString(fmt.Sprintf("ellipsis = %q\n", cfg.Style.Ellipsis))
	if len(cfg.Style.CustomInstructions) > 0 {
		b.WriteString("custom_instructions = [\n")
		for _, ci := range cfg.Style.CustomInstructions {
			b.WriteString(fmt.Sprintf("  %q,\n", ci))
		}
		b.WriteString("]\n")
	}
	return b.String()
}

// internal/tidy/quote.go
package tidy

import (
	"strings"
)

// QuotedBlock records a block of quoted lines and where it appeared.
type QuotedBlock struct {
	StartLine int
	Lines     []string
}

// isQuoted returns true if the line is a quoted line (starts with
// optional whitespace then ">").
func isQuoted(line string) bool {
	return strings.HasPrefix(strings.TrimLeft(line, " \t"), ">")
}

// SplitQuoted separates input text into author text and quoted blocks.
// Author text has quoted blocks replaced with blank lines to preserve
// paragraph structure. Quoted blocks record their original position.
func SplitQuoted(input string) (author string, quoted []QuotedBlock) {
	lines := strings.Split(input, "\n")
	// Remove trailing empty element from final newline
	if len(lines) > 0 && lines[len(lines)-1] == "" {
		lines = lines[:len(lines)-1]
	}

	var authorLines []string
	i := 0
	for i < len(lines) {
		if isQuoted(lines[i]) {
			block := QuotedBlock{StartLine: i}
			for i < len(lines) && isQuoted(lines[i]) {
				block.Lines = append(block.Lines, lines[i])
				i++
			}
			quoted = append(quoted, block)
			// Placeholder blank line in author text
			authorLines = append(authorLines, "")
		} else {
			authorLines = append(authorLines, lines[i])
			i++
		}
	}

	if len(authorLines) == 0 {
		return "", quoted
	}

	// Join and add trailing newline to match input convention
	result := strings.Join(authorLines, "\n")
	// Trim to check if only whitespace/blanks remain
	if strings.TrimSpace(result) == "" {
		return "", quoted
	}
	return result + "\n", quoted
}

// Reassemble combines corrected author text with the original quoted
// blocks in their original positions. It works line-by-line: author
// lines are pulled from corrected text, quoted blocks are inserted
// at their original positions.
func Reassemble(corrected, originalAuthor, originalInput string) string {
	if corrected == "" {
		return originalInput
	}

	origLines := strings.Split(originalInput, "\n")
	if len(origLines) > 0 && origLines[len(origLines)-1] == "" {
		origLines = origLines[:len(origLines)-1]
	}

	corrLines := strings.Split(corrected, "\n")
	if len(corrLines) > 0 && corrLines[len(corrLines)-1] == "" {
		corrLines = corrLines[:len(corrLines)-1]
	}

	var result []string
	ci := 0
	for _, line := range origLines {
		if isQuoted(line) {
			result = append(result, line)
		} else {
			if ci < len(corrLines) {
				result = append(result, corrLines[ci])
				ci++
			}
		}
	}
	// Append any extra corrected lines (model may have split a line)
	for ci < len(corrLines) {
		result = append(result, corrLines[ci])
		ci++
	}

	return strings.Join(result, "\n") + "\n"
}

// internal/tidy/prompt.go
package tidy

import "strings"

// BuildPrompt constructs the system prompt from the config's enabled
// rules and style preferences.
func BuildPrompt(cfg Config) string {
	var b strings.Builder

	b.WriteString("You are a proofreader. Fix errors in the text below and return ONLY the corrected text with no commentary, explanations, or markdown formatting.\n")

	// Build rules section
	var rules []string
	if cfg.Rules.Spelling {
		rules = append(rules, "Fix misspelled words")
	}
	if cfg.Rules.Grammar {
		rules = append(rules, "Fix grammar errors (to/too, its/it's, their/there/they're, affect/effect, then/than, lose/loose)")
	}
	if cfg.Rules.Punctuation {
		dash := "Convert dashes used as em dashes to the em dash character (\u2014)"
		if cfg.Style.EmDashSpaces {
			dash += " with spaces around them"
		} else {
			dash += " with no spaces around them"
		}
		rules = append(rules, dash)
		rules = append(rules, "Convert dashes used as en dashes to the en dash character (\u2013)")
		if cfg.Style.Ellipsis == "character" {
			rules = append(rules, "Convert three dots (...) to the ellipsis character (\u2026)")
		} else {
			rules = append(rules, "Convert the ellipsis character (\u2026) to three dots (...)")
		}
	}
	if cfg.Rules.Whitespace {
		rules = append(rules, "Fix whitespace errors (double spaces, trailing spaces)")
	}
	if cfg.Rules.Capitalization {
		rules = append(rules, "Fix capitalization errors (start of sentences, standalone \"I\")")
	}
	if cfg.Rules.RepeatedWords {
		rules = append(rules, "Fix repeated words (\"the the\", \"is is\")")
	}
	if cfg.Rules.MissingPunctuation {
		rules = append(rules, "Fix missing punctuation (missing period at end of final sentence, double commas)")
	}
	switch cfg.Rules.OxfordComma {
	case "insert":
		rules = append(rules, "Insert Oxford comma before the final item in lists of three or more")
	case "remove":
		rules = append(rules, "Remove Oxford comma before the final item in lists of three or more")
	}

	if len(rules) > 0 {
		b.WriteString("\nRules:\n")
		for _, r := range rules {
			b.WriteString("- ")
			b.WriteString(r)
			b.WriteString("\n")
		}
	}

	// Guardrails (always present)
	b.WriteString("\nDo NOT:\n")
	b.WriteString("- Rephrase or restructure sentences\n")
	b.WriteString("- Change tone or formality\n")
	b.WriteString("- Expand or contract contractions\n")
	b.WriteString("- Add or remove content\n")
	b.WriteString("- Change the author's voice or style\n")
	b.WriteString("- Modify text inside backtick code spans or code blocks\n")

	for _, ci := range cfg.Style.CustomInstructions {
		b.WriteString("- ")
		b.WriteString(ci)
		b.WriteString("\n")
	}

	b.WriteString("\nIf the text has no errors, return it exactly as-is.\n")

	return b.String()
}

// internal/tidy/api.go
package tidy

import (
	"bytes"
	"encoding/json"
	"fmt"
	"io"
	"net/http"
	"time"
)

const defaultAPIURL = "https://api.anthropic.com/v1/messages"

type apiRequest struct {
	Model     string       `json:"model"`
	MaxTokens int          `json:"max_tokens"`
	System    string       `json:"system"`
	Messages  []apiMessage `json:"messages"`
}

type apiMessage struct {
	Role    string `json:"role"`
	Content string `json:"content"`
}

type apiResponse struct {
	Content []contentBlock `json:"content"`
}

type contentBlock struct {
	Type string `json:"type"`
	Text string `json:"text"`
}

type apiError struct {
	Error struct {
		Message string `json:"message"`
	} `json:"error"`
}

// CallAPI sends text to the Claude API and returns the corrected text.
func CallAPI(url, apiKey, model, systemPrompt, text string) (string, error) {
	reqBody := apiRequest{
		Model:     model,
		MaxTokens: 4096,
		System:    systemPrompt,
		Messages: []apiMessage{
			{Role: "user", Content: text},
		},
	}

	body, err := json.Marshal(reqBody)
	if err != nil {
		return "", fmt.Errorf("marshaling request: %w", err)
	}

	req, err := http.NewRequest("POST", url, bytes.NewReader(body))
	if err != nil {
		return "", fmt.Errorf("creating request: %w", err)
	}
	req.Header.Set("content-type", "application/json")
	req.Header.Set("x-api-key", apiKey)
	req.Header.Set("anthropic-version", "2023-06-01")

	client := &http.Client{Timeout: 30 * time.Second}
	resp, err := client.Do(req)
	if err != nil {
		return "", fmt.Errorf("API unavailable: %w", err)
	}
	defer resp.Body.Close()

	respBody, err := io.ReadAll(resp.Body)
	if err != nil {
		return "", fmt.Errorf("reading response: %w", err)
	}

	if resp.StatusCode != http.StatusOK {
		var apiErr apiError
		json.Unmarshal(respBody, &apiErr)
		return "", fmt.Errorf("API error (%d): %s", resp.StatusCode, apiErr.Error.Message)
	}

	var apiResp apiResponse
	if err := json.Unmarshal(respBody, &apiResp); err != nil {
		return "", fmt.Errorf("parsing response: %w", err)
	}

	if len(apiResp.Content) == 0 {
		return "", fmt.Errorf("empty response from API")
	}

	return apiResp.Content[0].Text, nil
}

// internal/tidy/tidy.go
package tidy

import "strings"

// Status codes for the tidy result.
const (
	StatusCorrected    = iota // text was changed
	StatusNoChanges           // text had no errors
	StatusNoAuthorText        // input was all quoted text
	StatusError               // an error occurred, original text returned
)

// Result holds the output of a tidy operation.
type Result struct {
	Text    string // the (possibly corrected) text
	Status  int
	Message string // human-readable status for stderr
}

// Tidy processes input text: strips quoted lines, sends author text to
// the Claude API for correction, and reassembles the result. On any
// error, returns the original text unchanged with a descriptive message.
func Tidy(input string, cfg Config, apiKey, apiURL string) (Result, error) {
	if apiURL == "" {
		apiURL = defaultAPIURL
	}

	author, _ := SplitQuoted(input)

	if author == "" {
		return Result{
			Text:    input,
			Status:  StatusNoAuthorText,
			Message: "tidytext: no author text found",
		}, nil
	}

	if apiKey == "" {
		return Result{
			Text:    input,
			Status:  StatusError,
			Message: "tidytext: ANTHROPIC_API_KEY not set, text unchanged",
		}, nil
	}

	prompt := BuildPrompt(cfg)
	corrected, err := CallAPI(apiURL, apiKey, cfg.API.Model, prompt, author)
	if err != nil {
		return Result{
			Text:    input,
			Status:  StatusError,
			Message: "tidytext: " + err.Error() + ", text unchanged",
		}, nil
	}

	// Ensure corrected text ends with newline
	if !strings.HasSuffix(corrected, "\n") {
		corrected += "\n"
	}

	reassembled := Reassemble(corrected, author, input)

	// Compare to detect if anything changed
	if reassembled == input {
		return Result{
			Text:    input,
			Status:  StatusNoChanges,
			Message: "tidytext: no changes needed",
		}, nil
	}

	// Count corrections (simple line diff)
	origLines := strings.Split(input, "\n")
	newLines := strings.Split(reassembled, "\n")
	changes := 0
	maxLen := len(origLines)
	if len(newLines) > maxLen {
		maxLen = len(newLines)
	}
	for i := 0; i < maxLen; i++ {
		var o, n string
		if i < len(origLines) {
			o = origLines[i]
		}
		if i < len(newLines) {
			n = newLines[i]
		}
		if o != n {
			changes++
		}
	}

	noun := "correction"
	if changes != 1 {
		noun = "corrections"
	}

	return Result{
		Text:    reassembled,
		Status:  StatusCorrected,
		Message: "tidytext: " + itoa(changes) + " " + noun + " applied",
	}, nil
}

func itoa(n int) string {
	if n == 0 {
		return "0"
	}
	var digits []byte
	for n > 0 {
		digits = append([]byte{byte('0' + n%10)}, digits...)
		n /= 10
	}
	return string(digits)
}

		inputPath = args[0]
		data, err := os.ReadFile(inputPath)
		if err != nil {
			return fmt.Errorf("reading %s: %w", inputPath, err)
		}
		input = string(data)
	} else {
		// Check if stdin is a TTY (no pipe, no file)
		if term.IsTerminal(int(os.Stdin.Fd())) {
			return fmt.Errorf("no input provided (pipe text or pass a file argument)")
		}
		data, err := io.ReadAll(os.Stdin)
		if err != nil {
			return fmt.Errorf("reading stdin: %w", err)
		}
		input = string(data)
	}

	// Load config
	var cfg tidy.Config
	var err error
	if f.noConfig {
		cfg = tidy.DefaultConfig()
	} else {
		cfg, err = tidy.LoadConfig(f.configPath)
		if err != nil {
			fmt.Fprintf(os.Stderr, "tidytext: config error: %v\n", err)
			fmt.Fprint(os.Stdout, input)
			return nil
		}
	}

	// Apply overrides
	if len(f.rules) > 0 {
		if err := tidy.ApplyRuleOverrides(&cfg, f.rules); err != nil {
			return fmt.Errorf("rule override: %w", err)
		}
	}
	if len(f.styles) > 0 {
		if err := tidy.ApplyStyleOverrides(&cfg, f.styles); err != nil {
			return fmt.Errorf("style override: %w", err)
		}
	}

	apiKey := tidy.ResolveAPIKey(cfg)

	result, err := tidy.Tidy(input, cfg, apiKey, "")
	if err != nil {
		return err
	}

	// Print status to stderr
	fmt.Fprintln(os.Stderr, result.Message)

	// Write output
	if f.inPlace && inputPath != "" {
		dir := filepath.Dir(inputPath)
		tmp, err := os.CreateTemp(dir, ".tidytext-*")
		if err != nil {
			return fmt.Errorf("creating temp file: %w", err)
		}
		tmpPath := tmp.Name()
		if _, err := tmp.WriteString(result.Text); err != nil {
			tmp.Close()
			os.Remove(tmpPath)
			return fmt.Errorf("writing temp file: %w", err)
		}
		tmp.Close()
		if err := os.Rename(tmpPath, inputPath); err != nil {
			os.Remove(tmpPath)
			return fmt.Errorf("renaming temp file: %w", err)
		}
	} else {
		fmt.Fprint(os.Stdout, result.Text)
	}

	return nil
}

-- tidytext: run prose tidier on author's body text
local function run_tidy()
  local lines = vim.api.nvim_buf_get_lines(0, 0, -1, false)

  -- Find body start: first blank line after headers
  local body_start = 1
  local in_headers = false
  for i, line in ipairs(lines) do
    if line:match("^[A-Za-z-]+:") then
      in_headers = true
    elseif in_headers and line == "" then
      body_start = i + 1
      break
    end
  end

  -- Find body end: exclude signature (line starting with "-- ")
  local body_end = #lines
  for i = body_start, #lines do
    if lines[i] == "-- " then
      body_end = i - 1
      break
    end
  end

  if body_start > body_end then
    vim.notify("tidytext: no body text found", vim.log.levels.INFO)
    return
  end

  -- Save original lines for diff
  local original = {}
  for i = body_start, body_end do
    original[#original + 1] = lines[i]
  end

  -- Pipe body through tidytext fix
  local input = table.concat(original, "\n") .. "\n"
  local output_lines = vim.fn.systemlist("tidytext fix", input)

  -- If the command failed, notify and return
  if vim.v.shell_error ~= 0 then
    vim.notify("tidytext: command failed", vim.log.levels.WARN)
    return
  end

  -- Replace body lines with output
  vim.api.nvim_buf_set_lines(0, body_start - 1, body_end, false, output_lines)

  -- Word-level diff: highlight changes
  local ns = vim.api.nvim_create_namespace("tidytext_changes")
  vim.api.nvim_buf_clear_namespace(0, ns, 0, -1)

  for i, new_line in ipairs(output_lines) do
    local old_line = original[i] or ""
    if new_line ~= old_line then
      -- Find changed words by splitting and comparing
      local old_words = vim.split(old_line, "%s+")
      local new_words = vim.split(new_line, "%s+")
      local col = 0
      for j, nw in ipairs(new_words) do
        local ow = old_words[j] or ""
        local word_start = new_line:find(nw, col + 1, true)
        if word_start and nw ~= ow then
          vim.api.nvim_buf_set_extmark(0, ns, body_start - 1 + i - 1, word_start - 1, {
            end_col = word_start - 1 + #nw,
            hl_group = "EmailTidyChange",
          })
        end
        if word_start then
          col = word_start + #nw - 1
        end
      end
    end
  end

  -- Clear highlights on next edit
  local clear_id
  clear_id = vim.api.nvim_create_autocmd({ "TextChanged", "TextChangedI" }, {
    buffer = 0,
    once = true,
    callback = function()
      vim.api.nvim_buf_clear_namespace(0, ns, 0, -1)
    end,
  })

  -- Notify user
  local changed = 0
  for i, new_line in ipairs(output_lines) do
    if new_line ~= (original[i] or "") then
      changed = changed + 1
    end
  end
  if changed > 0 then
    vim.notify("tidytext: " .. changed .. " line(s) changed", vim.log.levels.INFO)
  else
    vim.notify("tidytext: no changes needed", vim.log.levels.INFO)
  end
end

vim.keymap.set("n", "<leader>t", run_tidy, { desc = "Tidy prose (tidytext)" })
