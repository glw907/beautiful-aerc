# Configuration v1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Land the v1 configuration story — `config.toml` with provider-first model, self-documenting template, password-cmd resolution at Connect time, friendly validation, OS-aware path discipline, and `poplar config` subcommands.

**Architecture:** TOML key `backend` becomes `provider`; the file is renamed `accounts.toml` → `config.toml`. The `Provider` struct gains `InsecureTLS`; `AccountConfig` gains `PasswordCmd` (deferred resolution). On first launch with no config, poplar emits a self-documenting template and exits 78 (`EX_CONFIG`). Schema validation produces friendly errors (account name, file path, "did you mean" Levenshtein). New `poplar config` subcommands cover init/check/path; the existing folder-discover command is renamed to `poplar config discover-folders`.

**Tech Stack:** Go 1.26, `github.com/BurntSushi/toml`, `github.com/spf13/cobra`. No new dependencies.

**Reference:** Spec at `docs/superpowers/specs/2026-05-02-config-design.md`. Pre-1.0 churn rule applies — refactor and rename freely; no compat shims for old key/file names beyond the single "rename your accounts.toml" hint.

**Branch policy:** Work commits directly to `master` per project convention (pre-1.0; no worktrees).

```go
// SPDX-License-Identifier: MIT

package config

import (
	"os"
	"path/filepath"
	"runtime"
)

// Source records where the config path came from.
type Source int

const (
	SourceFlag Source = iota
	SourceEnv
	SourceDefault
)

// Resolve returns the config-file path to use, plus how it was
// chosen. Precedence: --config flag, then $POPLAR_CONFIG, then the
// OS default.
//
// Linux/macOS default: ~/.config/poplar/config.toml.
// Windows default:     %APPDATA%\poplar\config.toml.
//
// macOS deliberately uses ~/.config/ rather than the OS-default
// ~/Library/Application Support/, matching the convention used by
// pass, nvim, tmux, and git.
func Resolve(flagPath string) (string, Source, error) {
	if flagPath != "" {
		return flagPath, SourceFlag, nil
	}
	if env := os.Getenv("POPLAR_CONFIG"); env != "" {
		return env, SourceEnv, nil
	}
	dir, err := defaultConfigDir()
	if err != nil {
		return "", SourceDefault, err
	}
	return filepath.Join(dir, "poplar", "config.toml"), SourceDefault, nil
}

func defaultConfigDir() (string, error) {
	switch runtime.GOOS {
	case "darwin", "linux", "freebsd", "openbsd", "netbsd":
		// Use XDG; on macOS this overrides Apple's default Application Support.
		if v := os.Getenv("XDG_CONFIG_HOME"); v != "" {
			return v, nil
		}
		home, err := os.UserHomeDir()
		if err != nil {
			return "", err
		}
		return filepath.Join(home, ".config"), nil
	default:
		return os.UserConfigDir()
	}
}
```

Add to `internal/mailimap/auth.go`:

```go
import (
	// ... existing imports ...
	"os/exec"
	"strings"
)

// resolvePassword returns the cleartext password for cfg. Inline
// Password wins; otherwise PasswordCmd is run via /bin/sh -c and
// stdout (trimmed) is the password. Returns an error if neither
// is set or the command fails.
func resolvePassword(cfg *config.AccountConfig) (string, error) {
	if cfg.Password != "" {
		return cfg.Password, nil
	}
	if cfg.PasswordCmd == "" {
		return "", errors.New("account has no password or password-cmd")
	}
	cmd := exec.Command("/bin/sh", "-c", cfg.PasswordCmd)
	out, err := cmd.Output()
	if err != nil {
		stderr := ""
		if ee, ok := err.(*exec.ExitError); ok {
			stderr = strings.TrimSpace(string(ee.Stderr))
		}
		if stderr != "" {
			return "", fmt.Errorf("password-cmd failed: %s", stderr)
		}
		return "", fmt.Errorf("password-cmd failed: %w", err)
	}
	return strings.TrimRight(string(out), "\n"), nil
}
```

```go
import (
	// ... existing ...
	"errors"
	"path/filepath"
)

// ErrFirstRun is returned by Load when the default config path
// did not exist and a fresh template was written. The caller
// should print a "created <path> — edit and run again" message
// and exit with status 78 (EX_CONFIG).
var ErrFirstRun = errors.New("first-run: template written")

// ErrOldAccountsToml is returned when the user has an old
// accounts.toml file (pre-1.0 carryover) and no config.toml.
var ErrOldAccountsToml = errors.New("old accounts.toml detected; rename to config.toml")

// Load resolves the config path and returns the parsed accounts.
// On first run with no flag/env override, it writes the template
// and returns ErrFirstRun.
func Load(flagPath string) ([]AccountConfig, error) {
	path, src, err := Resolve(flagPath)
	if err != nil {
		return nil, err
	}
	data, err := os.ReadFile(path)
	if err == nil {
		return ParseAccountsFromBytes(data)
	}
	if !os.IsNotExist(err) {
		return nil, err
	}
	if src != SourceDefault {
		return nil, fmt.Errorf("config file %s not found", path)
	}
	// Default path missing — check for legacy accounts.toml in the
	// same directory before writing the template.
	dir := filepath.Dir(path)
	legacy := filepath.Join(dir, "accounts.toml")
	if _, statErr := os.Stat(legacy); statErr == nil {
		return nil, fmt.Errorf("%w: found %s", ErrOldAccountsToml, legacy)
	}
	if err := os.MkdirAll(dir, 0o755); err != nil {
		return nil, fmt.Errorf("create config dir: %w", err)
	}
	if err := os.WriteFile(path, []byte(Template()), 0o600); err != nil {
		return nil, fmt.Errorf("write template: %w", err)
	}
	return nil, fmt.Errorf("%w: %s", ErrFirstRun, path)
}
```

```go
// SPDX-License-Identifier: MIT

package config

// suggestProvider returns the closest known-provider name to s
// within Levenshtein distance 2, or "" if no close match exists.
// Includes the imap/jmap fallbacks in the search space.
func suggestProvider(s string) string {
	if s == "" {
		return ""
	}
	candidates := []string{"imap", "jmap"}
	for k := range Providers {
		candidates = append(candidates, k)
	}
	bestName := ""
	bestDist := 3 // 3 is "too far"; we only suggest at distance ≤ 2
	for _, c := range candidates {
		d := levenshtein(s, c)
		if d < bestDist {
			bestDist = d
			bestName = c
		}
	}
	if bestDist > 2 {
		return ""
	}
	return bestName
}

func levenshtein(a, b string) int {
	if a == b {
		return 0
	}
	la, lb := len(a), len(b)
	if la == 0 {
		return lb
	}
	if lb == 0 {
		return la
	}
	prev := make([]int, lb+1)
	curr := make([]int, lb+1)
	for j := 0; j <= lb; j++ {
		prev[j] = j
	}
	for i := 1; i <= la; i++ {
		curr[0] = i
		for j := 1; j <= lb; j++ {
			cost := 1
			if a[i-1] == b[j-1] {
				cost = 0
			}
			curr[j] = min3(curr[j-1]+1, prev[j]+1, prev[j-1]+cost)
		}
		prev, curr = curr, prev
	}
	return prev[lb]
}

func min3(a, b, c int) int {
	m := a
	if b < m {
		m = b
	}
	if c < m {
		m = c
	}
	return m
}
```

```go
// SPDX-License-Identifier: MIT

package mailimap

import (
	"net"
	"strings"
)

// looksSelfHosted reports whether host appears to be a self-hosted
// or homelab address (RFC 1918 IPv4, IPv6 ULA fc00::/7, .local
// mDNS, or 127.x). Used to gate the "set insecure-tls = true if
// self-signed" hint on TLS errors.
func looksSelfHosted(host string) bool {
	if strings.HasSuffix(host, ".local") {
		return true
	}
	ip := net.ParseIP(host)
	if ip == nil {
		return false
	}
	if ip.IsLoopback() || ip.IsPrivate() {
		return true
	}
	return false
}
```

```go
func newConfigInitTemplateCmd() *cobra.Command {
	var force bool
	cmd := &cobra.Command{
		Use:          "init",
		Short:        "Write a fresh self-documenting config template",
		SilenceUsage: true,
		RunE: func(cmd *cobra.Command, args []string) error {
			path, _, err := config.Resolve("")  // honor $POPLAR_CONFIG / default
			if err != nil {
				return err
			}
			if !force {
				if _, statErr := os.Stat(path); statErr == nil {
					return fmt.Errorf("%s already exists; use --force to overwrite", path)
				}
			}
			if err := os.MkdirAll(filepath.Dir(path), 0o755); err != nil {
				return err
			}
			if err := os.WriteFile(path, []byte(config.Template()), 0o600); err != nil {
				return err
			}
			fmt.Fprintf(cmd.OutOrStdout(), "wrote %s\n", path)
			return nil
		},
	}
	cmd.Flags().BoolVar(&force, "force", false, "overwrite existing config file")
	return cmd
}
```
